CREATE TABLE `User` ( `Unique` INT NOT NULL AUTO_INCREMENT , `Username` VARCHAR(200) NOT NULL , `UUID` VARCHAR(200) NOT NULL , `Password_Hash` VARCHAR(200) NOT NULL , `Trys` INT NOT NULL DEFAULT '3' , `Group` VARCHAR(200) NOT NULL , PRIMARY KEY (`Unique`)) ENGINE = InnoDB;
CREATE TABLE `Absence_Dates` ( `Unique` INT NOT NULL AUTO_INCREMENT , `UUID` VARCHAR(200) NOT NULL , `From` DATE NOT NULL , `To` DATE NOT NULL , PRIMARY KEY (`Unique`)) ENGINE = InnoDB;
CREATE TABLE `Online_Times` ( `Unique` INT NOT NULL AUTO_INCREMENT , `UUID` VARCHAR(200) NOT NULL , `Login` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `Logout` on update CURRENT_TIMESTAMP TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `Active_Minutes` INT NOT NULL , PRIMARY KEY (`Unique`)) ENGINE = InnoDB;
CREATE TABLE `Groups` ( `Unique` INT NOT NULL AUTO_INCREMENT , `Name` VARCHAR(200) NOT NULL , `Priority` INT NOT NULL , `G_ID` VARCHAR(200) NOT NULL , PRIMARY KEY (`Unique`)) ENGINE = InnoDB;
CREATE TABLE `GroupPlayer` ( `Unique` INT NOT NULL AUTO_INCREMENT , `G_ID` VARCHAR(200) NOT NULL , `P_ID` VARCHAR(200) NOT NULL , PRIMARY KEY (`Unique`)) ENGINE = InnoDB;

SELECT User.Username, Groups.Name as Groupname, Groups.Priority FROM Groups, User, GroupPlayer WHERE GroupPlayer.G_ID = Groups.G_ID AND GroupPlayer.P_ID = User.UUID ORDER BY Groups.Priority
